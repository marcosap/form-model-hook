## Install

```bash
npm install --save form-model-hook
```

## Usage

```jsx
import React, { Component } from 'react'

import { useMyHook } from 'form-model-hook'

const Example = () => {
  const example = useMyHook()
  return (
    <div>{example}</div>
  )
}
```
